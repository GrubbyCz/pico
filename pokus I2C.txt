import board
import digitalio
import busio
import time

LED_PIN = board.GP15
I2C_ADDR = 0x08

led = digitalio.DigitalInOut(LED_PIN)
led.direction = digitalio.Direction.OUTPUT

i2c = busio.I2C(scl=board.GP27, sda=board.GP26)
while not i2c.try_lock():
    pass

print("Slave ready!")

while True:
    try:
        data = bytearray(1)
        i2c.readfrom_into(I2C_ADDR, data)
        led.value = bool(data[0])
    except OSError:
        pass  # slave čeká na master
    time.sleep(0.05)
